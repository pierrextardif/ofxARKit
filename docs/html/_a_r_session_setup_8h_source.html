135  static ARSession * generateNewSession(SFormat format){
136  ARSession * session = [ARSession new];
137 
138  auto state = format.getState();
139 
140 #if AR_FACE_TRACKING
141  // first check if we want face tracking and if it's supported.
142  // Currently unknown if this affects other possible tracking implementations since
143  // it has it's own configuration type.
144  if(state.useFaceTracking){
145 
146  if([ARFaceTrackingConfiguration isSupported]){
147  ARFaceTrackingConfiguration * config = [ARFaceTrackingConfiguration new];
148 
149  if(state.useLightEstimation){
150  config.lightEstimationEnabled = YES;
151  }
152 
153  config.worldAlignment = state.worldAlignment;
154 
155  if(state.delegateClass != NULL){
156  session.delegate = state.delegateClass;
157 
158  // note that audio is only available when you use a delegate class.
159  if(state.useAudio){
160  config.providesAudioData = YES;
161  }
162  }
163 
164  [session runWithConfiguration:config];
165 
166  return session;
167  }else{
168  NSLog(@"Unable to use face tracking configuration, defaulting to a more standard config.");
169  }
170  }
171 #endif
172 
173  // if face tracking is not available, should pass through to here where we
174  // figure out regular configuration, starting with determining if we can do plane detection.
175  if([ARWorldTrackingConfiguration isSupported]){
176 
177  ARWorldTrackingConfiguration * config = [ARWorldTrackingConfiguration new];
178 
179  if(state.usePlaneTracking){
180  config.planeDetection = state.planeDetectionType;
181  }
182 
183  if(state.useLightEstimation){
184  config.lightEstimationEnabled = YES;
185  }
186 
187  // image detection
188  if (@available(iOS 11.3, *)) {
189  if ( state.useImageDetection ){
190  // this looks crazier than it is!
191  // basically, it just passes in the name of your resource group
192  config.detectionImages = [ARReferenceImage referenceImagesInGroupNamed:[NSString stringWithUTF8String:state.imageBundleName.c_str()] bundle:[NSBundle mainBundle]];
193  }
194  }
195 
196  config.worldAlignment = state.worldAlignment;
197  if (@available(iOS 11.3, *)) {
198  if ( state.useAutoFocus ){
199  config.autoFocusEnabled = true;
200  }
201  }
202 
203  if (@available(iOS 11.3, *)) {
204  // WIP API: high res = default 1080p (first item in configs array)
205  if (state.useHighRes){
206  config.videoFormat = ARWorldTrackingConfiguration.supportedVideoFormats[0];
207 
208  // Backup is lowest res avail; right now seems to be 720p
209  } else {
210  config.videoFormat = ARWorldTrackingConfiguration.supportedVideoFormats.lastObject;
211  }
212  }
213 
214  if(state.delegateClass != NULL){
215  session.delegate = state.delegateClass;
216 
217  // note that audio is only available when you use a delegate class.
218  if(state.useAudio){
219  config.providesAudioData = YES;
220  }
221  }
222 
223  [session runWithConfiguration:config];
224 
225 
226  }else {
227  // if we can't do plane detection, switch to regular tracking.
228  AROrientationTrackingConfiguration * config = [AROrientationTrackingConfiguration new];
229  if(state.useLightEstimation){
230  config.lightEstimationEnabled = YES;
231  }
232 
233  config.worldAlignment = state.worldAlignment;
234 
235  if(state.delegateClass != NULL){
236  session.delegate = state.delegateClass;
237 
238  // note that audio is only available when you use a delegate class.
239  if(state.useAudio){
240  config.providesAudioData = YES;
241  }
242  }
243 
244  [session runWithConfiguration:config];
245 
246  }
247 
248 
249 
250 
251  return session;
252 
253  }
