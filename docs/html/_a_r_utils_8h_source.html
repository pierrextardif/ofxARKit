103  // Set the dimensions as appropriate depending on our orientation.
104  // Note that for some reason, and I'm not sure if it's an oF, IOS or mistake on my part, but the first time
105  // it enters this switch block, dimensions are off, so there is an nested if statement to try and fix that in
106  // some of the cases.
107  switch(UIDevice.currentDevice.orientation){
108  case UIDeviceOrientationFaceUp:
109  setWidthAndHeight();
110 
111  // if face up - we just assume portrait
112  dimensions.x = width;
113  dimensions.y = height;
114  break;
115 
116  case UIDeviceOrientationFaceDown:
117  setWidthAndHeight();
118 
119  // if face up - we just assume portrait
120  dimensions.x = width;
121  dimensions.y = height;
122  break;
123  case UIInterfaceOrientationUnknown:
124  // if unknown - we just assume portrait
125  dimensions.x = width;
126  dimensions.y = height;
127  break;
128 
129  // upside down registers, but for some reason nothing happens and there might be weirdness :/
130  // leaving this here anyways but probably best to just disable upsidedown portrait.
131  case UIInterfaceOrientationPortraitUpsideDown:
132  setWidthAndHeight();
133 
134  dimensions.x = width;
135  dimensions.y = height;
136 
137  if(width > height){
138  dimensions.x = height;
139  dimensions.y = width;
140  }else{
141  dimensions.x = width;
142  dimensions.y = height;
143  }

145 
146  case UIInterfaceOrientationPortrait:
147  setWidthAndHeight();
148 
149

152  dimensions.y = width;
153  }else{
154  dimensions.x = width;
155  dimensions.y = height;

212  // convert world xyz position to screen position.
213  // TODO maybe this is what we ought to use instead to factor in orientation? Found on 11/7/17
214  // https://developer.apple.com/documentation/arkit/arcamera/2923538-projectpoint?language=objc
215  static ofVec2f worldToScreen(ofPoint worldPoint,ofMatrix4x4 projection,ofMatrix4x4 view){
216 
217  ofVec4f p = ofVec4f(worldPoint.x, worldPoint.y, worldPoint.z, 1.0);
218 
219  p = p * view;
220  p = p * projection;
221 
222  p /= p.w;
223 
224  // convert coords to 0 - 1
225  p.x = p.x * 0.5 + 0.5;
226  p.y = p.y * 0.5 + 0.5;
227 
228  // convert coords to pixels
229  p.x *= ofGetWidth();
230  p.y *= ofGetHeight();
231  return ofVec2f(p.x, p.y);
232  }

235  static ofVec4f screenToWorld(ofVec3f position,ofMatrix4x4 projection,ofMatrix4x4 mvMatrix){
236  ofRectangle viewport(0,0,ofGetWindowWidth(),ofGetWindowHeight());
237 
238  ofVec4f CameraXYZ;
239  CameraXYZ.x = 2.0f * (position.x - viewport.x) / viewport.width - 1.0f;
240  CameraXYZ.y = 1.0f - 2.0f *(position.y - viewport.y) / viewport.height;
241  CameraXYZ.z = position.z;
242  CameraXYZ.w = -10.0;
243 
244  ofMatrix4x4 inverseCamera;
245  inverseCamera.makeInvertOf(mvMatrix * projection);
246 
247  return CameraXYZ * inverseCamera;
248  }
